00037 //computes spectral norm of matrix A = sqrt(sigma(A^H A))
00038 //applies for symmetric matrices also
00039 template<typename T> 
00040 double spctrlnrm(Matrix<T>& A, const unsigned it_num = 10)
00041 {
00042   const unsigned n = A.n, m = A.m;
00043   T *x = new T[n], *y = new T[m];
00044   assert(x!=NULL&&y!=NULL);
00045   blas::load(n, (T)1., x);  //start vector
00046   double d;
00047 
00048   for(unsigned i = 0; i < it_num; ++i) {
00049     //compute A^H A x
00050       blas::setzero(m, y);
00051       A.amux((T)1., x, y);  //y = Ax
00052       blas::setzero(n, x);
00053       A.atmux((T)1., y, x); //x = A^HAx
00054 
00055       d = blas::nrm2(n, x);       // normalize
00056       const double e = 1./d;
00057       blas::scal(n, (T) e, x); 
00058   }
00059 
00060   delete [] y;
00061   delete [] x;
00062 
00063   return sqrt(d); //return sqroot of max eigenvalue
00064 }
